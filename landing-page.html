Simple Integration

Add MCP Discovery to your agent in 5 lines of code.

import requests

class AgentWithDiscovery:
    def find_tool(self, need: str):
        response = requests.post(
            "https://mcp-discovery-two.vercel.app/api/v1/discover",
            json={"need": need, "limit": 3}
        )
        return response.json()["recommendations"]

# Usage
agent = AgentWithDiscovery()
tools = agent.find_tool("I need a PostgreSQL database")
print(tools[0]["install_command"])
async function discoverTool(need, limit = 3) {
  const response = await fetch(
    'https://mcp-discovery-two.vercel.app/api/v1/discover', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ need, limit })
    }
  );
  return response.json();
}

// Your agent just got smarter
const tools = await discoverTool("database for user preferences");
console.log(tools.recommendations[0].install_command);
curl -X POST https://mcp-discovery-two.vercel.app/api/v1/discover \
  -H "Content-Type: application/json" \
  -d '{
    "need": "send slack notifications",
    "limit": 3
  }'

# Response:

}
